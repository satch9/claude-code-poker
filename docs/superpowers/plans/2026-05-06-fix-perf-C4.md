- [ ] **Step 2 : Refactor `getTablePlayers`**

Remplacer le pattern par un `Promise.all` explicite. Code attendu :

```typescript
import { getAuthUserId } from "@convex-dev/auth/server";
import { sanitizePlayer } from "./shared/sanitize";

export const getTablePlayers = query({
  args: { tableId: v.id("tables") },
  handler: async (ctx, args) => {
    const callerId = await getAuthUserId(ctx);
    const players = await ctx.db
      .query("players")
      .withIndex("by_table", (q) => q.eq("tableId", args.tableId))
      .collect();
    return Promise.all(
      players.map(async (p) => {

        return { ...sanitizePlayer(p, callerId), user };
      })
    );
  },
});
```
